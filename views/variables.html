Diferencia entre variables VAR VS LET VS CONST

            
                var edad = 10
                var edad = 20
                console.log(edad)

                let edad = 10
                let edad = 20
                console.log(edad)

                let edad = 10
                edad = 20
                console.log(edad)
                
                const edad = 10
                const edad = 20
                console.log(edad)
